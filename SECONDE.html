import random
def deviner():
    nombre_secret = random.randint(1, 10)
    essais = 0
    while True:
        choix = int(input("Devine un nombre entre 1 et 10 : "))
        essais += 1
        if choix == nombre_secret:
            print("Bravo ! Trouvé en", essais, "essais.")
            break
        elif choix < nombre_secret:
            print("Trop petit.")
        else:
            print("Trop grand.")
deviner()

def ajouter_contact(nom, numero):
    try:
        with open("contacts.txt", "a") as f:
            f.write(f"{nom}:{numero}\n")
        print("Contact ajouté.")
    except:
        print("Erreur lors de l'ajout.")

def lister_contacts():
    try:
        with open("contacts.txt", "r") as f:
            for ligne in f:
                nom, numero = ligne.strip().split(":")
                print(f"{nom} => {numero}")
    except FileNotFoundError:
        print("Aucun contact trouvé.")

while True:
    print("\n1. Ajouter\n2. Lister\n3. Quitter")
    choix = input("Choix : ")
    if choix == "1":
        nom = input("Nom : ")
        numero = input("Numéro : ")
        ajouter_contact(nom, numero)
    elif choix == "2":
        lister_contacts()
    elif choix == "3":
        break
    else:
        print("Choix invalide.")
